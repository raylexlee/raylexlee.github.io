#!/usr/bin/env bash
for P in $(awk '{print $3;}' < People.txt | sort -u)
do
grep $P People.txt | awk '{print $4,$2,$1;}' | nl -n rz -w 3 -s " " > person/"$P".txt
[ -d text/$P ] || mkdir text/$P
awk '{print $1,$2;}' < person/"$P".txt > text/$P/coverparameters.txt
while read -r line
do
DATE=$(awk '{print $1;}' <<< $line)
TITLE=$(awk '{print $2;}' <<< $line)
EID=$(awk '{print $3;}' <<< $line)
EPISODES_URL='https://www.rthk.hk/radio/radio1/programme/People/episode/'"$EID"
curl -s -o temp.html $EPISODES_URL
dos2unix -q temp.html
echo $TITLE > text/"$P"/"$DATE".txt
cat temp.html | pup 'meta[name="description"] attr{content}' >> text/"$P"/"$DATE".txt
done < person/"$P".txt
done
rm temp.html
